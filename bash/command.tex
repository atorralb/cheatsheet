\subsection*{Numerics}
\textit{\say{Numeric expansion} is indicated by }\texttt{\$(( <expr> ))}\textit{ syntax, where }\texttt{<expr>}\textit{ allows a wide range of simple mathmatical expressions.}\\
\entry{35mm}{\$(( 365 - \$(date +\%j) ))}{wks to new year \Smiley}\\
\entry{45mm}{[ \textbackslash ( 2 -gt 2\textbackslash) \&\& \textbackslash ( 4 -le 1 \textbackslash) ]}{good}\\
\entry{45mm}{[ \$(( (3 > 2) \&\& (4 <= 1) )) = 1 ]}{better}\\
\entry{45mm}{(( (3 > 2) \&\& (4 <= 1) ))}{best}\\

\subsection*{Synopsis}


\begin{tabular}{@{}l l l l}
    |           & pipe          & >             & std out  \\
    <           & std in        & >{}>          & append out \\
    >|          & force out     & \textit{n}>|  & force to \textit{n} \\
    <>          & in \& out     & \textit{n}<>  & in, out to \textit{n} \\
    <{}<        & \say{here doc} & \textit{n}>  & out to FD \textit{n} \\
    <\textit{n} & FD in \textit{n} & \textit{n}>{}> & FD append \textit{n} \\ 
    \textit{n}>\&  & dup out    & \textit{n}<\& & dup in, FD \textit{n} \\
    \textit{n}>\&\textit{m} & \textit{n} follows \textit{m} & \textit{n}<\&\textit{m} & \textit{n} follows \textit{m}  \\
    \&>         & out \& err    & <\&-     & close std in \\
    >\&-        & close s-out   & \textit{n}>\&- & close \textit{n}-out \\
\end{tabular}
